# Essential BPF/BTF Kernel Configuration Options for Linux 6.18.20

## Core BPF Support
CONFIG_BPF=y                           # Enable BPF subsystem
CONFIG_BPF_SYSCALL=y                   # Enable bpf() system call
CONFIG_BPF_JIT=y                       # Enable BPF JIT compiler
CONFIG_BPF_JIT_ALWAYS_ON=y             # Force JIT compilation (security)
CONFIG_BPF_JIT_DEFAULT_ON=y            # Enable JIT by default

## BTF (BPF Type Format) Support
CONFIG_DEBUG_INFO_BTF=y                # Generate BTF debug info
CONFIG_DEBUG_INFO_BTF_MODULES=y        # BTF for kernel modules
CONFIG_PAHOLE_HAS_SPLIT_BTF=y          # Split BTF support (if pahole >= 1.21)

## BPF Program Types
CONFIG_BPF_EVENTS=y                    # BPF programs on kprobes, tracepoints, etc.
CONFIG_BPF_KPROBE_OVERRIDE=y           # Allow BPF to override kprobes
CONFIG_BPF_STREAM_PARSER=y             # BPF stream parser

## Networking BPF
CONFIG_NET_CLS_BPF=m                   # BPF-based packet classification
CONFIG_NET_ACT_BPF=m                   # BPF-based packet action
CONFIG_BPF_STREAM_PARSER=y             # Stream parser for BPF
CONFIG_LWTUNNEL_BPF=y                  # Lightweight tunnel BPF support
CONFIG_NETFILTER_XT_MATCH_BPF=m        # iptables BPF match

## XDP (eXpress Data Path)
CONFIG_XDP_SOCKETS=y                   # AF_XDP sockets
CONFIG_XDP_SOCKETS_DIAG=y              # XDP socket diagnostics

## BPF LSM (Linux Security Module)
CONFIG_BPF_LSM=y                       # BPF-based LSM (requires LSM framework)
CONFIG_LSM="landlock,lockdown,yama,integrity,bpf"  # Add bpf to LSM list

## cgroup BPF
CONFIG_CGROUP_BPF=y                    # Attach BPF programs to cgroups
CONFIG_SOCK_CGROUP_DATA=y              # Socket cgroup data

## BPF Infrastructure
CONFIG_BPF_LIRC_MODE2=y                # BPF for LIRC mode2 devices
CONFIG_BPFILTER=y                      # BPF-based packet filtering
CONFIG_BPFILTER_UMH=m                  # BPF filter user mode helper

## Tracing & Observability
CONFIG_FTRACE=y                        # Function tracer (required for BPF)
CONFIG_KPROBES=y                       # Kernel probes
CONFIG_KPROBE_EVENTS=y                 # Kprobe-based event tracing
CONFIG_UPROBE_EVENTS=y                 # Userspace probe events
CONFIG_TRACEPOINTS=y                   # Tracepoint support
CONFIG_PERF_EVENTS=y                   # Performance monitoring

## Debug Info (required for BTF)
CONFIG_DEBUG_INFO=y                    # Enable debug info
CONFIG_DEBUG_INFO_DWARF_TOOLCHAIN_DEFAULT=y  # Use toolchain default DWARF version

## Dependencies
CONFIG_KALLSYMS=y                      # Kernel symbol table
CONFIG_KALLSYMS_ALL=y                  # All symbols in kallsyms
CONFIG_MODULES=y                       # Loadable module support
CONFIG_HAVE_EBPF_JIT=y                 # Architecture has eBPF JIT
